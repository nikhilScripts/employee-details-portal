Step 2.2 Configure MCP Settings

Add the following configuration to cline_mcp_settings.json:

{
  "mcpServers": {
    "jira": {
      "command": "node",
      "args": ["C:\\Users\\YourName\\path\\to\\jira-full-server\\build\\index.js"],
      "env": {
        "JIRA_HOST": "https://your-instance.atlassian.net",
        "JIRA_EMAIL": "your-email@example.com",
        "JIRA_API_TOKEN": "your-jira-api-token"
      },
      "disabled": false,
      "autoApprove": []
    },
    "github": {
      "command": "cmd",
      "args": ["/c", "npx", "-y", "@modelcontextprotocol/server-github"],
      "env": {
        "GITHUB_PERSONAL_ACCESS_TOKEN": "ghp_your-token-here"
      },
      "disabled": false,
      "autoApprove": []
    }
  }
}
